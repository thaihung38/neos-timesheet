
- app:
    - Support: application wide support classes
    - Models: application wide model classes
    - Module:
        - {$module_name}
            - Api: module APIs which can be called from other modules or controllers
            - Block: module blocks which transforming data between Model and View
            - config: module configs merged into app
                - acl.php: list resource for ACL
                - injection.php: bind module APIs to concrete classes
            - Controllers: module controllers
            - database:
                - factories: module factory files
                - migrations: module migration files
            - Middleware
            - Models: all models which implement interfaces from Api folder
            - Providers: module providers
            - resources:
                - css: module css files
                - js: module js files
                - lang: json language
                - views: template (blade)
            - routes:
                - web.php: web routes
                - api.php: api routes
            - {$module_name}ServiceProvider.php: register module to application
    - UML:
        - *.svg: general UML
        - {$module_name}
            - *.svg: module UML / diagrams
            - *.uml: module UML / diagrams can be imported to UML reader
